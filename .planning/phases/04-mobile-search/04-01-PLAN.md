  **3. Add `handleFabFileSelect()` JS function** — In the `<script>` block, after the `doUpload` function. It reads the selected file's `name`, searches `cachedClientFiles` for a match by basename (`.split('/').pop()`), checks if already synced via `computeStatus()`, and calls `doUpload()` if applicable. If no match, show feedback "Not in sync dir" on the FAB button. If already synced, show feedback "Already synced" on the FAB button. Reset `input.value = ''` after processing.

  ```javascript
  function handleFabFileSelect(input) {
      var file = input.files[0];
      if (!file) return;
      var name = file.name;
      var match = cachedClientFiles.find(function(f) {
          return f.rel_path.split('/').pop() === name;
      });
      input.value = '';
      var fabBtn = document.getElementById('fab-btn');
      if (!match) {
          showFeedback(fabBtn, 'Not in sync dir', true);
          return;
      }
      var status = computeStatus(cachedLocalMap, cachedServerMap, match.rel_path);
      if (status === 'synced') {
          showFeedback(fabBtn, 'Already synced', false);
          return;
      }
      doUpload(match.rel_path, fabBtn);
  }
  ```
